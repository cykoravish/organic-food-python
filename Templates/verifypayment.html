from django.http import JsonResponse from django.views.decorators.csrf import csrf_exempt import razorpay @csrf_exempt def verify_payment(request): if request.method == "POST": data = request.POST razorpay_payment_id = data.get('razorpay_payment_id') razorpay_order_id = data.get('razorpay_order_id') razorpay_signature = data.get('razorpay_signature') # Verify Razorpay signature params_dict = { 'razorpay_order_id': razorpay_order_id, 'razorpay_payment_id': razorpay_payment_id, 'razorpay_signature': razorpay_signature } try: client.utility.verify_payment_signature(params_dict) return JsonResponse({'status': 'success', 'message': 'Payment Verified'}) except razorpay.errors.SignatureVerificationError: return JsonResponse({'status': 'failed', 'message': 'Payment Verification Failed'})
